Install nvm, Node.js, and npm

```bash
sudo apt install curl -y 
sudo apt update
sudo apt -y upgrade

# Install nvm
curl https://raw.githubusercontent.com/creationix/nvm/master/install.sh | bash 
source ~/.bashrc

# Install Node.js and npm
nvm install node 
sudo npm install -y -g create-react-app react-scripts
```
